    b.HasKey(u => u.Id);

    // Indexes for "normalized" username and email, to allow efficient lookups

    b.HasIndex(u => u.NormalizedEmail).HasName("EmailIndex");

    b.ToTable("AspNetUsers");

    b.Property(u => u.ConcurrencyStamp).IsConcurrencyToken();
